selection_sort(N, arr):
    for i in range(N):
        min_value <- 9999
        min_idx <- 0
        for j in range(i+1,N):
            if arr[j] < min_value:
                min_idx <- j
                min_value <- arr[j]
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
